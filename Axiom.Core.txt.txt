using Axiom.Core;
using Axiom.Core.Contracts;
using Microsoft.AspNetCore.Http;

var builder = WebApplication.CreateBuilder(args);

// Bind explicitly. Prefer ListenAnyIP + firewall allowlist.
builder.WebHost.ConfigureKestrel(k =>
{
    k.ListenAnyIP(7071, listen =>
    {
        listen.UseHttps(); // use a real cert later; don't leave this ambiguous
    });
});

var app = builder.Build();

var sharedSecret = Environment.GetEnvironmentVariable("AXIOM_CORE_TOKEN")
    ?? throw new InvalidOperationException("AXIOM_CORE_TOKEN not set");

string dbPath = Path.Combine(AppContext.BaseDirectory, "axiom.db");
var core = new CoreEngine(dbPath);

static IResult Unauthorized() => Results.Unauthorized();

bool RequireAuth(HttpRequest req)
{
    return req.Headers.TryGetValue("X-Axiom-Token", out var token) && token == sharedSecret;
}

app.MapGet("/health", () => Results.Ok(new
{
    ok = true,
    service = "Axiom.Core.Service",
    pid = Environment.ProcessId,
    started_utc = DateTimeOffset.UtcNow
}));

app.MapPost("/index", (HttpRequest req, IndexRequest body) =>
{
    if (!RequireAuth(req)) return Unauthorized();
    var runId = core.RunIndex(body.TargetRoot);
    return Results.Ok(new { run_id = runId });
});

app.MapPost("/bundle", (HttpRequest req, BundleRequest body) =>
{
    if (!RequireAuth(req)) return Unauthorized();
    var result = core.RunBundle(body);
    return Results.Json(result);
});

app.MapPost("/append-derived", (HttpRequest req, DerivedFactRequest body) =>
{
    if (!RequireAuth(req)) return Unauthorized();
    core.AppendDerived(body);
    return Results.Ok();
});

app.Run();
